Script To run on website with save:

var iframecr = document.createElement("iframe") iframecr.src = "https://alexidians.github.io/" document.body.appendChild(iframecr) const iframe = document.querySelector("iframe") const wind = iframe.contentWindow const data = { password: localStorage.getItem("Password") background: localStorage.getItem("background") backgroundIsImage: localStorage.getItem("backgroundIsImage") startupProgram: localStorage.getItem("startupProgram") } wind.postMessage(JSON.stringify(data), "*")
